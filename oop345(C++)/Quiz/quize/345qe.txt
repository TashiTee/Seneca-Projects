3- Having the following two functions:
void foo(int value);
void faa(double value);
write a program to run the above two functions simultaneously passing 1234 to the first
function and 123.45 to the second one.
Make sure all the threads are joined at the end.

#include <iostream>       // std::cout
#include <thread>         // std::thread, std::this_thread::sleep_for

void foo(int value) {
    std::cout << "foo " << value << "\n";
}
void faa(double value) {
     std::cout << "faa " << value << "\n";
}
 
int main() 
{
  std::cout << "Spawning 2 threads...\n";
  std::thread t1 (foo,1234);
  std::thread t2 (faa,123.45);

  std::cout << "Done spawning threads. Now waiting for them to join:\n";
 
  t1.join();
  t2.join();

  std::cout << "All threads joined!\n";

  return 0;
}
